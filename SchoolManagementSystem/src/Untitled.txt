    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
         /* JFileChooser chooser = new JFileChooser();

    if(returnVal == JFileChooser.APPROVE_OPTION) 
    {
     String fn = chooser.getSelectedFile().getAbsolutePath();     */
 
        try {
            FileDialog fc = new FileDialog(this, "Choose a file", FileDialog.LOAD);
            fc.setVisible(true);
            String fn = fc.getFile();
            if (fn != null)
            {
                FileInputStream fis = new FileInputStream(fn);
                ObjectInputStream ois = new ObjectInputStream(fis);
                products = (ArrayList<Product>) ois.readObject(); 
                fis.close();
                ois.close();
                
                 DefaultTableModel model =(DefaultTableModel) jTable1.getModel();
                 model.setRowCount(products.size()); 
                int row=0;
                int col;
                double total = 0;

                for (Product p:products)
                {     col =0;
                      model.setValueAt(p.getDescription(),row ,col++);
                      model.setValueAt(p.getPrice(),row ,col++);
                      model.setValueAt(p.getUnits(),row ,col);
                      total+= p.getPrice()*p.getUnits();
                      row++;
                }
                jTextField1.setText(""+total);


                       }
                   } catch (Exception ex) {
                   }           

    
     
    }//GEN-LAST:event_jButton1ActionPerformed
